// Ex 19.1 — npm setup

npm install bootstrap sass // custom/styles.scss @use "sass:map"; // Ex 19.2 — Override Bootstrap variables BEFORE importing $primary: #f5a623; // amber instead of blue $border-radius: 0.75rem; // rounder corners everywhere $font-family-base: 'Bitter', serif; // Then import Bootstrap @import "../node_modules/bootstrap/scss/bootstrap"; // Your custom rules after .eventCard { border-color: $primary; border-radius: $border-radius; }
